RENDERDERS：自定义模板辅助方法

灵活的模板扩展方法！

为组件模板渲染提供增强方法


    Keystone.RENDERERS = {
        xx: {
            yy: function(context) {
                //参数context 当前实例
                return "xx_" + context.get('name') + "_yy"
            }
        }
    };

模板中对应的渲染增加方法为"{{id_xx_yy}}"


    <div id="container1">
    </div>
    <script type="text/template" id="tmpl_keystone1">
        <div id="keystone1">
            <span>Hello
                <span>{{keystone1_xx_yy}}</span>
            </span>
        </div>
    </script>

使用过程中,调用render方法渲染


    KISSY.use('components/Keystone/',function(S,Keystone){
        var keystone = new Keystone({container:'#container1',tmpl:S.one('#tmpl_keystone1').html()});
        keystone.render();
    });
